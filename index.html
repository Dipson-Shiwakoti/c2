Other Preprocessor Directives

CODE

            #include <stdio.h>
            #define PI 3.14159
            #define AREA_CIRCLE(r) (PI * (r) * (r))
            #define TEMP 100
            #pragma message "Compiling the program..."
            
            int main() {
                printf("Value of PI: %f\n", PI);
                
                float radius = 5.0;
                printf("Area of circle with radius %.1f: %.2f\n", radius, AREA_CIRCLE(radius));
                
                printf("Value of TEMP: %d\n", TEMP);
                
                #undef TEMP
                
                // Line control directive
                #line 100 "custom_file.c"
                printf("This line appears to be from line %d in %s\n", __LINE__, __FILE__);
                
                return 0;
            }

Formatted I/O: printf() and scanf()

CODE

            #include <stdio.h>

            int main() {
                int age;
                float height;
                char name[50];
                
                // Using scanf() to get input
                printf("Enter your name: ");
                scanf("%s", name);
                
                printf("Enter your age: ");
                scanf("%d", &age);
                
                printf("Enter your height (in meters): ");
                scanf("%f", &height);
                
                // Using printf() with different format specifiers
                printf("\n----- Personal Information -----\n");
                printf("Name: %s\n", name);         // %s for string
                printf("Age: %d years\n", age);     // %d for integer
                printf("Height: %.2f meters\n", height); // %.2f for float with 2 decimal places
                printf("Age in hex: %X\n", age);    // %X for hexadecimal
                printf("Name with padding: %20s\n", name); // %20s for right-aligned string
                
                return 0;
            }

OUTPUT

Unformatted I/O: getchar(), putchar(), gets(), puts()

CODE

            #include <stdio.h>

            int main() {
                char ch;
                char str[100];
                
                // getchar() and putchar()
                printf("Enter a character: ");
                ch = getchar();
                printf("You entered: ");
                putchar(ch);
                printf("\n");
                
                // Clear input buffer
                while ((getchar()) != '\n');
                
                // gets() and puts()
                printf("Enter a string: ");
                gets(str);  // Note: gets() is unsafe and deprecated
                
                printf("You entered: ");
                puts(str);  // puts() automatically adds a newline
                
                return 0;
            }

Switch Case and Goto Label

CODE

            #include <stdio.h>

            int main() {
                int choice;

                printf("Enter a number between 1 and 3: ");
                scanf("%d", &choice);

                switch(choice) {
                    case 1:
                        printf("You chose 1\n");
                        break;
                    case 2:
                        printf("You chose 2\n");
                        break;
                    case 3:
                        printf("You chose 3\n");
                        break;
                    default:
                        printf("Invalid choice\n");
                }

                int num = 5;
                if (num > 3) {
                    goto label;
                }

                printf("This will be skipped.\n");

                label:
                    printf("Goto label reached.\n");

                return 0;
            }

Different Patterns

Square Pattern

CODE

            #include 

            int main() {
                int i, j;
                for (i = 0; i < 5; i++) {
                    for (j = 0; j < 5; j++) {
                        printf("* ");
                    }
                    printf("\n");
                }
                return 0;
            }

OUTPUT

Rectangle Pattern

CODE

            #include <stdio.h>

            int main() {
                int rows = 4, cols = 6;
                int i, j;
                
                for (i = 0; i < rows; i++) {
                    for (j = 0; j < cols; j++) {
                        printf("* ");
                    }
                    printf("\n");
                }
                return 0;
            }

OUTPUT

Right Pyramid Pattern

CODE

            #include <stdio.h>

            int main() {
                int rows = 5;
                int i, j;
                
                for (i = 0; i < rows; i++) {
                    for (j = 0; j <= i; j++) {
                        printf("* ");
                    }
                    printf("\n");
                }
                return 0;
            }

OUTPUT

Inverse Right Pyramid Pattern

CODE

            #include <stdio.h>

            int main() {
                int rows = 5;
                int i, j;
                
                for (i = rows; i > 0; i--) {
                    for (j = 0; j < i; j++) {
                        printf("* ");
                    }
                    printf("\n");
                }
                return 0;
            }
